country_event = {
	id = kuat_origin.1
	hide_window = yes
	is_triggered_only = yes
	trigger = { has_origin = origin_kuat_apocalyptic }
	immediate = {
		every_planet_within_border = {
			limit = { has_planet_flag = ring_habitat_kuat_world }
			set_name = "Name_kuat_system_1"
			add_deposit = d_str_kuat_spring
			while = { count = 1 add_district = district_rw_science }
			while = { count = 1 add_district = district_rw_city }
		}
	}
}
