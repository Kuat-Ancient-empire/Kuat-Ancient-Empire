			create_fleet = {
				settings = {

						target = prev

						angle = random
					}
				}
			}
		}
		create_message = {
			type = "SPACEPORT_CONSTRUCTION_MESSAGE_TYPE"

			target = from
			variable = {
				type = name
				localization = "capital_planet"
				scope = root.capital_scope
			}
		}

			which = fallen_empire_escort_queue

		}
