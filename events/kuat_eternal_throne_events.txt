country_event = {
	id = kuat_EEstart.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_origin = origin_eternalthrone
		NOT = { has_country_flag = new_coruscant_deposit }
	}
	immediate = {
		set_country_flag = Eternalempire
		set_country_flag = new_coruscant_deposit
		random_owned_planet = {
			limit = { has_planet_flag = EE_Home_World }
			remove_all_districts = yes
			clear_deposits = yes
			while = { count = 3 add_district = district_arcology_housing }
			while = { count = 3 add_district = district_arcology_arms_industry }
			add_deposit = ss_auto_deposits
			add_deposit = ss_Solarfarm
			add_deposit = ss_volatile_motes
			add_deposit = ss_crystallinemin
			add_deposit = ss_gaseous_eruption
			set_name = "Zakuul"
			solar_system = { set_name = "Zakuul" }
		}
		change_country_flag = {
			icon = {
				category = "exe_invasion"
				file = "ge_o.dds"
			}
			background = {
				category = "backgrounds"
				file = "00_solid.dds"
			}
			colors = { "yellow" "black" "null" "null" }
		}
	}
}
