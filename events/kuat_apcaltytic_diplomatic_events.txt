country_event = {
	id = kuat_diplomatic.14
	title = "kuat_diplomatic.14.title"
	desc = "kuat_diplomatic.14.desc"
	diplomatic = yes
	picture_event_data = {
		portrait = root.ruler
		room = galactic_custodian_room
	}
	custom_gui = "kuat_event_diplomacy_window"
	custom_gui_option = "kuat_event_diplomacy_option"
	is_triggered_only = yes
	fire_only_once = yes
	trigger = { is_player_country = yes }
	immediate = {
		change_variable = { which = eternal_bornus_var value = 150 }
		random_system = {
			limit = { has_star_flag = myst_system_5 }
			if = {
				limit = { event_target:kuat_player_target_country = { has_ethic = ethic_gestalt_consciousness } }
				clear_global_event_target = kuat_player_target_country
				clear_global_event_target = kuat_player_target_species
				random_country = {
					limit = { NOT = { has_ethic = ethic_gestalt_consciousness } is_country_type = default }
					save_global_event_target_as = kuat_player_target_country
					owner_main_species = { save_global_event_target_as = kuat_player_target_species }
				}
			}
			create_country = {
				name = "NAME_Kuat_E_Area_Guardian"
				type = Kuat_F_Area_Guardian
				species = event_target:kuat_player_target_species
				flag = {
					icon = {
						category = "Kuat"
						file = "flag_swsit_1.dds"
					}
					background = {
						category = "backgrounds"
						file = "00_solid.dds"
					}
					colors = { "red" "red" "null" "null" }
				}
				effect = {
					set_country_flag = protected_from_queen_storm
					add_modifier = { modifier = Elternal_creating }
					set_graphical_culture = fallen_empire_kuat_01
					set_country_flag = Kuat_F_Area_Guardian
					save_global_event_target_as = Kuat_F_Area_Guardian
					create_ship_design = { design = "NAME_KUAT_Starbase" }
					add_ship_design = last_created_design
					establish_communications_no_message = root
				}
			}
			create_starbase = {
				owner = event_target:Kuat_F_Area_Guardian
				size = exe_starbase
			}
			Kuat_F_Area_Guardian_fleet_spawn = yes
		}
	}
	option = {
		name = kuat_diplomatic.14.a
		response_text = kuat_diplomatic.14.a.res
	}
}
