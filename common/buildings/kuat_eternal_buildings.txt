building_eternal_alloys_Forge = {
	base_buildtime = 480
	
	base_cap_amount = 1
	category = manufacturing
	
	potential = {
		exists = owner
		OR = {
			owner = { is_player_country = yes }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
			has_origin = origin_eternalthrone
		}
	}
	
	allow = {
		has_upgraded_capital = yes
		exists = owner
		NOT = {
			has_building = building_eternal_business_complex
			has_building_construction = building_eternal_business_complex
		}
	}
	
	destroy_trigger = {
		free_building_slots = 0
		exists = owner
		owner = {
			is_ai = yes
			has_monthly_income = {
				resource = minerals
				value > 500
			}
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner				
		}
		modifier = {
			job_Kuat_foundry_add = 3
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 800
		}	
		upkeep = {
			energy = 25
		}
	}	
	
	prerequisites = {
		"tr_eternal_building_alloys_Forge"
	}
	
	triggered_desc = {
		text = job_foundry_effect_desc
	}

}

building_eternal_business_complex = {
	base_buildtime = 480

	base_cap_amount = 1
	category = manufacturing

	potential = {
		exists = owner
		owner = {
			OR = {
				is_gestalt = no
				has_valid_civic = civic_machine_servitor
			}
			has_origin = origin_eternalthrone
		}
		NOT = { has_modifier = resort_colony }
		NAND = {
			owner = { is_ai = yes }	
			has_any_industry_district = no
		}
	}

	allow = {
		exists = owner
		NOT = { 
			has_building = building_eternal_alloys_Forge 
			has_building_construction = building_eternal_alloys_Forge
		}
	}

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony

			AND = {
				exists = owner
				owner = {
					is_ai = yes
					is_gestalt = yes
					NOT = { has_valid_civic = civic_machine_servitor }
				}
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
		}
		modifier = {
			job_Kuat_artisan_add = 3
		}
	}
    planet_modifier = {
		planet_jobs_consumer_goods_produces_mult = 0.01
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = 750
		}
		upkeep = {
			energy = 25
		} 
	}

	prerequisites = {
		"tr_eternal_building_business_complex"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_artisan_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_artisan_effect_desc
	}

	ai_resource_production = {
		consumer_goods = 1
		trigger = {
			always = yes
		}
	}
}
