#科研
building_kuat_command_center_auto = {
	base_buildtime = 480
	capital = yes
	can_build = no
	can_demolish = no
	can_be_ruined = no
	can_be_disabled = no
	position_priority = 0
	exempt_from_ai_planet_specialization = yes
	category = government
	icon = building_kuat_command_center
	potential = {
		exists = owner
		owner = { has_technology = tech_kuat_command_center }
	}
	destroy_trigger = {
		always = no
	}
	building_sets = {
		automation
	}
	triggered_planet_modifier = {
		potential = { always = yes }
		show_only_custom_tooltip = yes
		custom_tooltip = "building_kuat_command_center_desc"
		pop_amenities_usage_mult = -1
		pop_amenities_usage_no_happiness_mult = -1
		pop_housing_usage_mult = -1
		planet_crime_mult = -1
		planet_crime_add = -100
		planet_amenities_add = 10000
		planet_stability_add = 100
	}
	convert_to = {
		building_kuat_command_center_auto
	}
	triggered_planet_modifier = {
		potential = { has_cosmic_storms_dlc = yes }
		planet_storm_devastation_mult = -1
	}

	inline_script = {
		script = cosmic_storms/StormBuildingModifiers
	}

	on_built = {
		if = {
			limit = { has_cosmic_storms_dlc = yes }
			solar_system = {
				set_star_flag = storm_no_negative_effects
				remove_modifier = storm_base_modifier
			}
		}
	}

	on_destroy = {
		if = {
			limit = { has_cosmic_storms_dlc = yes }
			owner = { set_country_flag = shield_on_destroy } #an action when destroyed
			solar_system = {
				remove_star_flag = storm_no_negative_effects
			}
		}
	}
	triggered_country_modifier = {
		potential = { exists = owner owner = { is_ai = no } }
		country_naval_cap_add = 1
		country_admin_cap_add = 1
		country_resource_max_add = 10
		mult = root.planet.planet_production_array
	}
	triggered_planet_modifier = {
		potential = { 
			exists = owner 
			owner = { NOR = {
				is_robot_empire = yes
				is_hive_empire = yes
			} }
		}
		job_kuat_auto_machine_job_per_pop = 1
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { OR = {
				is_robot_empire = yes
				is_hive_empire = yes
			} }
		}
		job_kuat_auto_machine_drone_per_pop = 1
	}
	triggered_planet_modifier = {
		potential = { exists = owner owner = { is_ai = no } }
		pop_happiness = 1
		planet_orbital_bombardment_damage = -0.01
		mult = planet.planet_production_array
	}
	upgrades = { }
}
