initialize_Rakata_crisis_system_select = {
    optimize_memory
	every_system = {
		limit = { 
			OR = {
				any_country = {
					OR = {
						is_country_type = fallen_empire
						is_country_type = awakened_fallen_empire
					}
					any_system_within_border = { is_same_value = PREVPREV }
				}
				any_country = {
					is_country_type = default
					capital_scope = { solar_system = { is_same_value = PREVPREVPREV } }
				}
			}
		}
		set_timed_star_flag = { flag = forbit_spawn_system days = 1 }
	}
	every_system = {
		limit = {
			NOT = { has_star_flag = forbit_spawn_system }
			OR = {
				kuat_is_kuat_system	= yes
				closest_system = {
					max_steps = 3
					limit = { has_star_flag = forbit_spawn_system }
				}
			}
		}
		set_timed_star_flag = { flag = forbit_spawn_system days = 1 }
	}
}

	}

kuat_infinite_create_clone_fleet_effect = {
    optimize_memory
    fromfrom = {
        every_owned_ship = {
            save_event_target_as = ifinite_copy_ship
            if = {
                limit = { NOT = { exists = event_target:kuat_battle_infinite_created_@prev } }
                create_fleet = {
                    name = prev
                    effect = {
                        set_owner = prevprev.owner
                        create_ship = { 
                            design = event_target:ifinite_copy_ship 
                            effect = { copy_flags_and_variables_from = event_target:ifinite_copy_ship }
                        }
                        save_global_event_target_as = kuat_battle_infinite_created_@prevprev
                        set_location = { target = prevprev distance = 45 angle = 0 }
                    }
                }
            }
            else = { event_target:kuat_battle_infinite_created_@prev = { create_ship = { 
                design = event_target:ifinite_copy_ship 
                effect = { copy_flags_and_variables_from = event_target:ifinite_copy_ship }
            } } }
        }
        clear_global_event_target = kuat_battle_infinite_created_@this
        set_fleet_flag = kuat_infinite_fleet_created_agained
    }
}
