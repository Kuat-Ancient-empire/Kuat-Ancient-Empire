		custom_tooltip_fail = {
			text = "Building_The_Kuat_Command_Line_Auto_planet_failed"
			OR = { 
				AND = { from = { is_ai = no } PR_prmt_trgr_HC = { MORE = 9 } }
				AND = { from = { is_ai = yes } has_global_flag = enable_ai_trigger }
			}
		}
	}
	effect = { hidden_effect = { kuat_automatic_ai_colony = yes } }
	ai_weight = { 
		weight = 1
		modifier = { factor = 0 OR = { has_global_flag = disable_ai_trigger has_global_flag = acot_activated } }
		modifier = { factor = 150000 years_passed >= 25 has_global_flag = enable_ai_trigger }
	}
}
