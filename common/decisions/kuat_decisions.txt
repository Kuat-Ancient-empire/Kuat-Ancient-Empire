Building_The_Kuat_Command_Line = {
	owned_planets_only = yes
	sound = event_air_raid_siren
	icon = decision_arcology_project
	resources = {
		category = decisions
		cost = { minerals = 5000 }
	}
	potential = {
		OR = {
			num_pops >= 75
			AND = {
				is_variable_set = z_pr_var_plnt_pop_man
				check_variable = { which = z_pr_var_plnt_pop_man value >= 75 }
			}
		}
		from = { has_technology = tech_kuat_command_center }
		OR = {
			has_building = building_system_capital
			has_building = building_nirvana_empire_palace
		}
		NOT = { has_building = building_kuat_command_center }
	}
	effect = {
		custom_tooltip = "building_kuat_command_center"
		if = {
			limit = { has_building = building_system_capital }
			remove_building = building_system_capital
			add_building = building_kuat_command_center
		}
		else_if = {
			limit = { has_building = building_nirvana_empire_palace }
			remove_building = building_nirvana_empire_palace
			add_building = building_kuat_command_center
		}
	}
	ai_weight = { weight = 0 }
}
