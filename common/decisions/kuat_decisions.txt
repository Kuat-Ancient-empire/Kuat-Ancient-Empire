Building_The_Kuat_Command_Line = {
	owned_planets_only = yes
	sound = event_air_raid_siren
	icon = decision_arcology_project
	resources = {
		category = decisions
		cost = { minerals = 5000 }
	}
	potential = {
		OR = {
			num_pops >= 100
			AND = {
				is_variable_set = z_pr_var_plnt_pop_man
				check_variable = { which = z_pr_var_plnt_pop_man value >= 100 }
			}
		}
		from = { has_technology = tech_kuat_command_center }
		OR = {
			has_building = building_system_capital
			has_building = building_nirvana_empire_palace
		}
		NOT = { has_building = building_kuat_command_center }
	}
	effect = {
		custom_tooltip = "building_kuat_command_center"
		if = {
			limit = { has_building = building_system_capital }
			remove_building = building_system_capital
			add_building = building_kuat_command_center
		}
		else_if = {
			limit = { has_building = building_nirvana_empire_palace }
			remove_building = building_nirvana_empire_palace
			add_building = building_kuat_command_center
		}
	}
	ai_weight = { weight = 0 }
}

Building_The_Kuat_Shield = {
	owned_planets_only = yes
	sound = event_air_raid_siren
	icon = decision_planet_beam
	enactment_time = 30
	resources = {
		category = decisions
		cost = { energy = 100 }
	}
	potential = {
		has_building = building_kuat_shield
		kuat_is_ringworld = no
		kuat_is_habitat = no
		NOT = { has_planet_flag = enable_kuat_shield }
		NOT = { has_planet_flag = fallen_empire_world }
	}
	effect = {
		custom_tooltip = "building_kuat_shield_tooltip"
		hidden_effect = { 
			create_ambient_object = {
				type = kuat_planet_shield_effect_object
				entity_scale_to_size = yes
				use_3d_location = yes
				entity_offset = { min = 0 max = 0 }
				entity_offset_angle = { min = 0 max = 0 }
				entity_offset_height = { min = 0 max = 0 }
				scale = 0.125
				location = this
				effect = { 
					set_ambient_object_flag = kuat_planet_shield_effect_object_@root 
				}
			}
			set_planet_flag = enable_kuat_shield
		}
	}
	ai_weight = { weight = 0 }
}

Destroy_The_Kuat_Shield = {
	owned_planets_only = yes
	sound = event_air_raid_siren
	icon = decision_planet_beam

	resources = {
		category = decisions
		cost = { energy = 100 }
	}

		has_building = building_kuat_shield
		has_planet_flag = enable_kuat_shield
	}
	effect = {
		custom_tooltip = "Destroy_The_Kuat_Shield_tooltip"
		hidden_effect = {

					limit = {
						has_ambient_object_flag = kuat_planet_shield_effect_object_@root

				}
			}
			remove_planet_flag = enable_kuat_shield
		}
	}
	ai_weight = { weight = 0 }
