swfo_supermacy = {
	entity = "supermacy_fixable_frame_entity"
	construction_entity = "supermacy_fixable_frame_entity"
	portrait = "GFX_megastructure_construction_background"	
	place_entity_on_planet_plane = no
	show_galactic_map_icon = no	
	show_in_outliner = no
	build_time = 15	
	upgrade_desc = hide	
	entity_offset = { x = 0 y = 0 }	
	upgrade_from = { swfo_supermacy_skeletion }	
	resources = {
		category = megastructures
		cost = { alloys = 500000 }
	}	
	potential = { has_country_flag = can_build_supermacy }
	possible = {
		custom_tooltip = {
			fail_text = "kuat_requires_no_existing_station"
			from = { NOT  = {
				any_owned_megastructure = { is_megastructure_type = swfo_supermacy }
				any_owned_ship = { is_ship_size = swfo_supremacy }
			} }
		}
	}	
	placement_rules = {}	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			remove_planet_flag = has_exe_starbase
			solar_system = {
				set_star_flag = kuat_starbase_megastructure_around_system
				set_star_flag = kuat_starbase_megastructure_in_system
				every_system_planet = {
					limit = {
						exists = owner
						owner = { NOT = { is_same_value = from } }
					}
					add_modifier = { modifier = "exe_megastructure_starbase_in_system" }
				}
				add_modifier = { modifier = "kuat_starbase_megastructure_around_system" }
			}
		}
		create_ambient_object = { type = "contingency_1" location = this }
		every_system = {
			limit = {
				NOT = { is_same_value = prev }
				distance = {
					source = prev 
					type = euclidean
					min_distance >= 0
					max_distance <= 25
				}
				NOT = { has_star_flag = kuat_starbase_megastructure_around_system }
			}
			set_star_flag = kuat_starbase_megastructure_around_system
			add_modifier = { modifier = "kuat_starbase_megastructure_around_system" }
			create_ambient_object = { type = "contingency_1" location = this }
		}
		add_threat = { who = from amount = 50 }
		create_fleet = {
			name = "NAME_EXE_SWFO_SUPERMACY"
			settings = { spawn_debris = no can_disband = no }
			effect = {
				set_fleet_flag = kuat_station_player
				set_owner = from
				create_ship = { 
					design = "NAME_EXE_SWFO_SUPERMACY" 
					effect = { 
						set_ship_flag = kuat_station_player 
						set_ship_flag = enable_disable_starbase 
						set_disable_at_health = 0.01
					}
				}
				set_location = { target = fromfrom distance = -1 angle = 0 }
				set_fleet_flag = ag_kuat_invasion_starbase_disable
				set_fleet_flag = ag_kuat_invasion_starbase_player_optimize
			}
		}
		from = {
			save_global_event_target_as = build_the_exe_eternal_starbase
			every_owned_starbase = {
				limit = { has_starbase_size = swfo_supremacy }
				set_starbase_building = { slot = 1 building = titan_yards }
				set_starbase_building = { slot = 2 building = colossus_yards }
			}
		}
		remove_megastructure = fromfrom
	}
	construction_blocks_and_blocked_by = self_type	
	ai_weight = { weight = 5000 }
}
