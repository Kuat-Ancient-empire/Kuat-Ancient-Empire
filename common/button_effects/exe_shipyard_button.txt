#shipyard
kuat_set_shipyard_add_horriwor = {
	potential = { from = { is_scope_type = country has_country_flag = kuat_build_fifth_ships has_country_flag = building_the_horriwer NOT = { has_country_flag = build_all_sell_ship_for_design_fleet } } }
	allow = { from = {
		hidden_trigger = { is_scope_type = country }
		custom_tooltip = { fail_text = kuat_shipyard.check get_ship_alloys_allow = yes }
	} }
	effect = {
		custom_tooltip = "exe_shipyard.a"
		hidden_effect = { from = { set_country_flag = fired_HSD change_variable = { which = HSD_queue value = event_target:global_event_country.kuat_shipyard_buillding_queue_selection } get_alloys_variable = { BUY = yes } } }
	}
}

kuat_set_fleet_shipyard_add_horriwor = {
	potential = { from = { is_scope_type = country has_country_flag = kuat_build_fifth_ships has_country_flag = building_the_horriwer has_country_flag = build_all_sell_ship_for_design_fleet } }
	allow = { from = {
		hidden_trigger = { is_scope_type = country }
		custom_tooltip = { fail_text = kuat_shipyard.check get_ship_alloys_allow = yes }
		custom_tooltip = { fail_text = kuat_shipyard.uppon_fleet check_variable = { which = kuat_shipyard_exist_commandline_UI value < 4000 } }
	} }
	effect = {
		custom_tooltip = "exe_shipyard.a"
		hidden_effect = { from = { 
			set_country_flag = fired_HSD 
			change_variable = { which = HSD_queue value = event_target:global_event_country.kuat_shipyard_buillding_queue_selection } 
			set_variable = { which = mulf_HSD_queue value = HSD_queue }
			multiply_variable = { which = mulf_HSD_queue value = 16 }
			set_variable = { which = kuat_shipyard_exist_commandline_UI value = value:kuat_shipyard_exist_commandline }
			get_alloys_variable = { BUY = yes } 
		} }
	}
}
