#shipyard
kuat_set_shipyard_add_horriwor = {
	potential = { from = { is_scope_type = country has_country_flag = kuat_build_fifth_ships has_country_flag = building_the_horriwer NOT = { has_country_flag = build_all_sell_ship_for_design_fleet } } }
	allow = { from = {
		hidden_trigger = { is_scope_type = country }
		custom_tooltip = { fail_text = kuat_shipyard.check get_ship_alloys_allow = yes }
	} }
	effect = {
		custom_tooltip = "exe_shipyard.a"
		hidden_effect = { from = { set_country_flag = fired_HSD change_variable = { which = HSD_queue value = event_target:global_event_country.kuat_shipyard_buillding_queue_selection } get_alloys_variable = { BUY = yes } } }
	}
}
