kuat_physics_auto_tech = {
	potential = {
		hidden_trigger = {
			from = {
				OR = {
					kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_energy_fire_rate }
					kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_energy_damage }
					kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_explosive_damage }
					kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_explosive_fire_rate }
				}
				NOR = {
					kuat_research_has_tech_name_continue = { TECH = tech_repeatable_weapon_type_energy_fire_rate }
					kuat_research_has_tech_name_continue = { TECH = tech_repeatable_weapon_type_energy_damage }
					kuat_research_has_tech_name_continue = { TECH = tech_repeatable_weapon_type_explosive_damage }
					kuat_research_has_tech_name_continue = { TECH = tech_repeatable_weapon_type_explosive_fire_rate }
				}
			}
		}
	}
	allow = { hidden_trigger = { always = yes } }
	effect = {
		hidden_effect = { from = {
			if = {
				limit = { kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_energy_damage } }
				kuat_research_set_tech_name_continue = { TECH = tech_repeatable_weapon_type_energy_damage }
			}
			if = {
				limit = { kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_energy_fire_rate } }
				kuat_research_set_tech_name_continue = { TECH = tech_repeatable_weapon_type_energy_fire_rate }
			}
			if = {
				limit = { kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_explosive_damage } }
				kuat_research_set_tech_name_continue = { TECH = tech_repeatable_weapon_type_explosive_damage }
			}
			if = {
				limit = { kuat_research_has_tech_name = { TECH = tech_repeatable_weapon_type_explosive_fire_rate } }
				kuat_research_set_tech_name_continue = { TECH = tech_repeatable_weapon_type_explosive_fire_rate }
			}
		} }
	}
}
