kuat_epsilon_planet_killer_weapon_selector_trigger = {
	if = {
		limit = { hidden_trigger = { root = { NOT = { has_origin = origin_eternalthrone } } } }
		OR = {
			hidden_trigger = { root = { NOT = { has_country_flag = kuat_enabled_epsilon_planet_killer_selector } } }
			AND = {
				custom_tooltip = {
					fail_text = "kuat_requires_a_star_star_killer" 
					event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { NOR = { is_star_class = sc_kuat has_star_flag = kuat_military_base_system_home } }
				}
				custom_tooltip = {
					fail_text = "kuat_requires_outside_border_star_killer"
					event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { NOT = { is_inside_border = root } }
				}
				custom_tooltip = {
					fail_text = "kuat_requires_no_kuat_system" 
					event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { 
						kuat_is_kuat_system = no 
						NOR = { 
							has_star_flag = ag_kae_e_area_system
							has_star_flag = monolith_crisis_systems
							has_star_flag = monolith_crisis_shield_system
							has_star_flag = eternal_starbase_megastructure_in_system 
						}
					}
				}
				custom_tooltip = {
					fail_text = "kuat_requires_no_ancient_system" 
					event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { 
						ag_is_ancient_system = no 
						NOT = { any_ship_in_system = { OR = {
							is_ship_size = ag_gamma_station_hostile
							is_ship_size = ag_gamma_station_small_hostile
							is_ship_size = ag_gamma_station_small_moving_hostile
						} } }
					}
				}
				custom_tooltip = {
					fail_text = "kuat_requires_no_zeta_crisis_system" 
					event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = {
						NOT = { any_ship_in_system = { OR = {
								is_ship_size = ag_zeta_crisis_portal
								is_ship_size = ag_zeta_crisis_psionic_storm
								is_ship_size = ag_zeta_crisis_psionic_singularity
						} } }
					}
				}
				custom_tooltip = {
					fail_text = "kuat_requires_no_exd_system" 
					event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { NOT = {
						any_ship_in_system = { is_ship_size = dimensional_portal_ed }
					} }
				}
				custom_tooltip = {
					fail_text = "kuat_requires_no_contingency_system" 
					event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { NOT = {
						any_system_planet = {
							is_planet_class = pc_ai
							exists = owner
							owner = { is_country_type = ai_empire }
						}
					} }
				}
				custom_tooltip = {
					fail_text = "kuat_requires_no_too_many_targets" 
					OR = {
						root = {
							exists = event_target:ag_kuat_invasion_country
							has_global_flag = difficult_exe_invasion
							check_variable = { which = kuat_num_system_selector_selected_systems value < 10 }
						}
						kuat_system_selector_system_selected_trigger = { kuat_system_seq = $kuat_system_seq$ }
						root = { check_variable = { which = kuat_num_system_selector_selected_systems value < 5 } }
					}
				}
			}
		}
	}
	else = { OR = {
		hidden_trigger = { root = { NOT = { has_country_flag = kuat_enabled_epsilon_planet_killer_selector } } }
		AND = {
			custom_tooltip = {
				fail_text = "kuat_requires_outside_border_star_killer"
				event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { NOT = { is_inside_border = root } }
			}
			custom_tooltip = {
				fail_text = "kuat_requires_declares_war_star_killer"
				event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { OR = {
					is_inside_border = root
					AND = { exists = space_owner space_owner = { is_hostile = root } }
					AND = { NOT = { exists = space_owner } any_fleet_in_system = { owner = { is_hostile = root } } }
				} }
			}
			custom_tooltip = {
				fail_text = "kuat_requires_no_too_many_targets" 
				OR = {
					kuat_system_selector_system_selected_trigger = { kuat_system_seq = $kuat_system_seq$ }
					root = { check_variable = { which = kuat_num_system_selector_selected_systems value < event_target:eternal_support_country.count_eternal_center_target } }

			custom_tooltip = {

				event_target:kuat_system_selector_marked_system_[[kuat_system_seq]$kuat_system_seq$] = { NOT = { has_star_flag = ag_kae_e_area_system } }
			}
