##GET KUAT SHIPYARD ALLOYS
kuat_shipyard_check_allow_resource = {
	optimize_memory
	resource_stockpile_compare = {
		resource = alloys
		value >= @\[ $second_alloys$ * $basic$ ]
		multiplier = value:kuat_shipyard_value
	}
}

get_ship_alloys_allow = {
	optimize_memory
	if = {
		limit = { has_country_flag = building_the_horriwer }
		if = {
			limit = { NOT = { has_country_flag = kuat_build_fifth_ships } }
			kuat_shipyard_check_allow_resource = { second_alloys = @building_the_horriwer_cost basic = $basic$ }
		}
		if = {
			limit = { has_country_flag = kuat_build_fifth_ships }
			kuat_shipyard_check_allow_resource = { second_alloys = @building_the_horriwer_cost basic = $basic$ }
		} 
	}
	else_if = {
		limit = { has_country_flag = building_the_empire } 
		if = {
			limit = { NOT = { has_country_flag = kuat_build_fifth_ships } }
			kuat_shipyard_check_allow_resource = { second_alloys = @building_the_empire_cost basic = $basic$ }
		}
		if = {
			limit = { has_country_flag = kuat_build_fifth_ships }
			kuat_shipyard_check_allow_resource = { second_alloys = @building_the_empire_cost basic = $basic$ }
		}
	}
	else_if = {
		limit = { has_country_flag = building_the_resugrant }
		if = {
			limit = { NOT = { has_country_flag = kuat_build_fifth_ships } }
			kuat_shipyard_check_allow_resource = { second_alloys = @building_the_resugrant_cost basic = $basic$ }
		}
		if = {
			limit = { has_country_flag = kuat_build_fifth_ships }
			kuat_shipyard_check_allow_resource = { second_alloys = @building_the_resugrant_cost basic = $basic$ }
		} 
	}
	else_if = {
		limit = { has_country_flag = building_the_donimance }
		kuat_shipyard_check_allow_resource = { second_alloys = @building_the_donimance_cost basic = $basic$ }
	}
	else_if = {
		limit = { has_country_flag = building_the_eclipse }
		kuat_shipyard_check_allow_resource = { second_alloys = @building_the_eclipse_cost basic = $basic$ }
	}
	else_if = {
		limit = { has_country_flag = building_the_Vengeance }
		kuat_shipyard_check_allow_resource = { second_alloys = @building_the_Vengeance_cost basic = $basic$ }
	}
}

get_variable_alloys_check_allow = {
	get_ship_alloys_allow = { basic = 1 }
}
